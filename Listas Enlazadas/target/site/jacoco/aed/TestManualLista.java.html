package aed;
public class TestManualLista {
    public static void main(String[] args) {
        ListaEnlazada<Integer> lista = new ListaEnlazada<>();

        // Agrego elementos
        lista.agregarAtras(42);
        lista.agregarAtras(43);
        lista.agregarAtras(44);

        // Creo el iterador
        Iterador<Integer> it = lista.iterador();

        // Test paso a paso
        System.out.println("¿Hay siguiente? " + it.haySiguiente()); // true
        System.out.println("¿Hay anterior? " + it.hayAnterior());   // false

        System.out.println("Siguiente: " + it.siguiente());         // 42
        System.out.println("¿Hay anterior? " + it.hayAnterior());   // false (no avanzó aún)

        System.out.println("Siguiente: " + it.siguiente());         // 43
        System.out.println("¿Hay anterior? " + it.hayAnterior());   // true

        System.out.println("Anterior: " + it.anterior());           // 42
        System.out.println("¿Hay anterior? " + it.hayAnterior());   // false

        System.out.println("¿Hay siguiente? " + it.haySiguiente()); // true
        System.out.println("Siguiente: " + it.siguiente());         // 43

        System.out.println("¿Hay siguiente? " + it.haySiguiente()); // true
        System.out.println("Siguiente: " + it.siguiente());         // 44

        System.out.println("¿Hay siguiente? " + it.haySiguiente()); // false

        System.out.println("¿Hay anterior? " + it.hayAnterior());   // true
        System.out.println("Anterior: " + it.anterior());           // 43

        System.out.println("¿Hay anterior? " + it.hayAnterior());   // true
        System.out.println("Anterior: " + it.anterior());           // 42

        System.out.println("¿Hay anterior? " + it.hayAnterior());   // false
    }
}
